        <p class="tip-body">Grind beans no more than 30 minutes before brewing. Ground coffee loses 60% of its aromatics within 15 minutes of exposure to air.</p>

      </div>

      <div class="tip-card">

        <div class="tip-icon">🌡️</div>

        <h3 class="tip-title">Water Temperature</h3>

        <p class="tip-body">Ideal brewing temperature is 90–96 °C (195–205 °F). Boiling water scorches the grounds; too cool and you'll under-extract, leaving it sour and weak.</p>

      </div>

      <div class="tip-card">

        <div class="tip-icon">💧</div>

        <h3 class="tip-title">Water Quality</h3>

        <p class="tip-body">Filtered water with a TDS of 75–150 ppm is optimal. Tap water with chlorine masks subtle flavours; distilled water produces flat, lifeless coffee.</p>

      </div>

      <div class="tip-card">

        <div class="tip-icon">⏱️</div>

        <h3 class="tip-title">Extraction Time</h3>

        <p class="tip-body">Espresso: 25–30 seconds. Pour-over: 3–4 minutes. French press: 4 minutes. Timing is the difference between a balanced cup and a bitter or sour one.</p>

      </div>

      <div class="tip-card">

        <div class="tip-icon">🫙</div>

        <h3 class="tip-title">Storage</h3>

        <p class="tip-body">Store whole beans in an airtight, opaque container at room temperature. Avoid the fridge — moisture and odour absorption ruin flavour profiles.</p>

      </div>

      <div class="tip-card">

        <div class="tip-icon">🔢</div>

        <h3 class="tip-title">The Golden Ratio</h3>

        <p class="tip-body">Start with 1 gram of coffee per 15 grams of water and adjust to taste. Precision scales are a brewer's most valuable tool after the grinder.</p>

      </div>

    </div>

// Modal

const overlay = document.getElementById('modalOverlay');

const modalClose = document.getElementById('modalClose');


function openModal(i) {

  const r = recipes[i];

  document.getElementById('modalTag').textContent = r.tag;

  document.getElementById('modalTitle').textContent = r.name;

  document.getElementById('modalDesc').textContent = r.desc;

  document.getElementById('modalIngredients').innerHTML = r.ingredients

    .map(ing => `<li>${ing.name}<span>${ing.amount}</span></li>`).join('');

  document.getElementById('modalSteps').innerHTML = r.steps

    .map(s => `<li>${s}</li>`).join('');

  document.getElementById('modalTip').textContent = r.tip;

  overlay.classList.add('open');

  document.body.style.overflow = 'hidden';

}


function closeModal() {

  overlay.classList.remove('open');

  document.body.style.overflow = '';

}


modalClose.addEventListener('click', closeModal);

overlay.addEventListener('click', e => { if (e.target === overlay) closeModal(); });

document.addEventListener('keydown', e => { if (e.key === 'Escape') closeModal(); });

  entries.forEach(e => {

    if (e.isIntersecting) {

      e.target.style.opacity = '1';

      e.target.style.transform = 'translateY(0)';

    }

  });

}, { threshold: 0.1 });
